arbol_data_tree <- Node$new("P")
primer_nodo_hijo = Node$new("Ap")
segundo_nodo_hijo = Node$new("Ss")
arbol_data_tree$AddChildNode(primer_nodo_hijo)
arbol_data_tree$AddChildNode(segundo_nodo_hijo)


tercer_nodo_hijo = Node$new("Ss")
tree <- Node$new("T")
tree$AddChildNode(arbol_data_tree)
tree$AddChildNode(tercer_nodo_hijo)

primer_nodo_hijo$label = c("A", "B")
segundo_nodo_hijo$label = c("C", "D")
tercer_nodo_hijo$label = c("D")
arbol_data_tree$label = c("A", "B", "C")

SetEdgeStyle(primer_nodo_hijo, label = primer_nodo_hijo$label, fontsize = 7)
SetEdgeStyle(segundo_nodo_hijo, label = segundo_nodo_hijo$label, fontsize = 7)
SetEdgeStyle(tercer_nodo_hijo, label = tercer_nodo_hijo$label, fontsize = 7)
SetEdgeStyle(arbol_data_tree, label = arbol_data_tree$label, fontsize = 7)

plot(tree)

print(tree, "label")
